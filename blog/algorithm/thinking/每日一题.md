## 001. 斐波那契数列

`question` 输入一个整数，请输出斐比那契数列的第n项

`answer`
``` js
// 1. 递归解法
function Fibonacci(n) {
    if(n < 2) { return n }
    return Finonacci(n-1) + Fibonacci(n-2)
}

// 2.  递归加记忆化
function Fibonacci(n, memory) {
    if(n < 2) { return n }
    if(!memory[n]) {  memory[n] = Fibonacci(n-1, memory) + Fibonacci(n-2, memory) }
    return memory[n]
}

// 3. 动态规划解法
function Finonacci(n) {
    if(n<2 ) { return n }
    let i = 1, prev = 0, current = 1,result = 0
    while(i++ < n) {
        result = prev + current;
        prev = current;
        current = result
    }
    return current
}
```

## 003. 字符串排列

`question` 输入一个字符串，按字典打印出该字符串中字符的所有排列<br>
例如输入一个字符串abc，则打印出a,b,c能排列出路啊的所有字符串abc,acb,bac,bca,cab,cba

`提示`
使用回溯法
1. 记录一个字符串(temp)，用于存储当前需要进入排列的字符串
2. 记录一个字符串(current),用于记录当前已经排列好的字符
3. 记录一个队列(queue)，用于存储还未被排列的字符

- 每次排列将tmp添加到current
- 如果queue为空，则本次排列完成，将current加入到结果数组中，结束递归；如果queue不为空，说明还有为排列的字符
- 递归排列queue中剩余的字符

`answer`
``` js
// 解法1. 记录当前排列字符串temp
function Permutation(str) {
    const result = []
    if(str) {
        queue = str.split('')
        PermutationCore(queue,result)
    }
    result.sort()
    return [...new Set(result)]
}

function PermutationCore(queue, result, tmp = '', current= '') {
    current += tmp
    if(queue.length === 0) {
        result.push(current) 
        return;
    }
    for(let i = 0; i < queue.length; i++) {
        tmp = queue.shift()
        PermutationCore(queue, result, tmp, current)
        queue.push(tmp)
    }
}
```
